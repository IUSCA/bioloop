    dataset ||--|| dataset_create_log : "has one"
    dataset_create_log ||--o| upload_log : "has one"
    upload_log ||--o{ file_upload_log : "has many"

    dataset {
        Int id PK
        String name
        String type
        Boolean is_deleted
    }

    dataset_create_log {
        Int id PK
        Int dataset_id FK,UK
        DATASET_CREATE_METHOD create_method
        Int creator_id FK
        Int upload_log_id FK,UK
    }

    upload_log {
        Int id PK
        Int dataset_create_log_id FK,UK
        upload_status status
    }

    file_upload_log {
        Int id PK
        String name
        String md5
        Int num_chunks
        upload_status status
        String path
        Int upload_log_id FK
    }
