    dataset_audit {
        Int id PK
        String action
        DateTime updated_at
        DateTime timestamp
        Json old_data
        Json new_data
        Int user_id FK
        Int dataset_id FK
    }

    dataset_upload_log {
        Int id PK
        upload_status status
        DateTime updated_at
        Int audit_log_id FK,UK
    }

    file_upload_log {
        Int id PK
        String name
        String md5
        Int num_chunks
        upload_status status
        String path
        Int dataset_upload_log_id FK
    }

    dataset {
        Int id PK
        String name
        String type
        Boolean is_deleted
        DATASET_CREATE_METHOD create_method
    }
